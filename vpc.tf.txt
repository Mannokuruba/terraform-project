resource "aws_vpc" "mannovpc" {
  cidr_block       = "${var.vpc_cidr}"
  instance_tenancy = "default"
tags = {
    Name = "manno VPC"
  }
}


# Creating 1st subnet 
resource "aws_subnet" "mannosubnet1" {
  vpc_id                  = "${aws_vpc.mannovpc.id}"
  cidr_block             = "${var.subnet1_cidr}"
  map_public_ip_on_launch = true
  availability_zone = "us-west-2a"
tags = {
    Name = "manno subnet 1"
  }
}
